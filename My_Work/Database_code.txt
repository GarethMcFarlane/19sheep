-- tables

-- Table commitments
CREATE TABLE commitments (
    id int  AUTO_INCREMENT,
    targetperday varchar(255),
    startday date,
    deadline date,
    text varchar(100),
    isachieved int,
    type char(50),
    email varchar(255),
    CONSTRAINT commitments_pk PRIMARY KEY (id)
);

-- Table message
CREATE TABLE message (
    id int,
    convoid int,
    title varchar(256),
    user1 varchar(256),
    user2 varchar(256),
    messages text,
    mytimestamp timestamp,
    user1read int,
    user2read int,
    CONSTRAINT message_pk PRIMARY KEY (id)
);

-- Table mood
CREATE TABLE mood (
    id int  AUTO_INCREMENT,
    useremail varchar(255),
    timestamp timestamp,
    val char(2),
    CONSTRAINT mood_pk PRIMARY KEY (id)
);

-- Table users
CREATE TABLE users (
    OAuthToken varchar(80),
    OAuthTokenSecret varchar(80),
    password varchar(64),
    fullName varchar(200),
    height decimal(2,1),
    weight decimal(3,1),
    gender char(1),
    birthday date,
    location varchar(50),
    salt varchar(50),
    email varchar(255),
    userid int,
    mood_id int,
    message_id int,
    CONSTRAINT users_pk PRIMARY KEY (email)
);
