CREATE TABLE activity (
    activityidpk int PRIMARY KEY,
    intenseact int,
    mediumact int,
    lightact int,
    numofsteps int,
    weight int,
    blooppressure int,
);
CREATE TABLE commitments (
    commitmentsidpk int PRIMARY KEY,
    activityidfk int,
    sleepidfk int,
    targetperday int,
    duration date,
    text char(100),
    isachieved bool,
FOREIGN KEY (sleepidfk) REFERENCES sleep (sleepidpk);
FOREIGN KEY (activityidfk) REFERENCES activity (activityidpk);
);
CREATE TABLE sleep (
    sleepidpk int PRIMARY KEY,
    deepsleep int,
    totalsleep int,
    wakeups int,
    eveningsleep int,
    morningsleep int,
    sleepscore int,
);
CREATE TABLE users (
    OAuthToken varchar(80),
    OAuthTokenSecret varchar(80),
    username varchar(64) PRIMARY KEY,
    password varchar(64),
    fullName varchar(200),
    height decimal(2,1),
    weight decimal(3,1),
    gender char(1),
    birthday date,
    location varchar(50),
    salt varchar(50),
    email varchar(255),
    userid int,
    commitmentsfk int,
    FOREIGN KEY (commitmentsfk) REFERENCES commitments (commitmentsidpk);
);
